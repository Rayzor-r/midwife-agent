## Objective

Add a full-screen password overlay to `static/index.html` that gates app access behind the API key stored in `localStorage`, wires `Authorization: Bearer` headers into all four `fetch()` call sites, and handles 401 / network-failure error states.

## must_haves

### goal
Every live API endpoint requires a valid bearer token, the CORS policy admits only the intended origin, the OAuth callback no longer exposes credentials, and the previously exposed Google credentials have been replaced.

### truths
- `static/index.html` contains `<div id="auth-overlay"` with `position:fixed` and `z-index:9999`
- `static/index.html` contains `getApiKey()` function returning `localStorage.getItem("api_key") || ""`
- `static/index.html` contains `"Authorization": "Bearer " + getApiKey()` at all four fetch() call sites: `/api/chat` (line ~597), `/api/upload` (line ~835), `/api/documents` GET (line ~849), `/api/documents/${id}` DELETE (line ~885)
- `static/index.html` contains `signOut()` function that calls `localStorage.clear()` and `location.reload()`
- `static/index.html` contains a 401 handler in the fetch wrappers that calls `localStorage.removeItem("api_key")` and re-shows the overlay with error text

```javascript
// ── Auth ─────────────────────────────────────────────────────────────────────

function getApiKey() {
  return localStorage.getItem('api_key') || '';
}

function showAuthOverlay(errorMsg) {
  const overlay = document.getElementById('auth-overlay');
  const errEl   = document.getElementById('auth-error');
  overlay.classList.remove('hidden');
  errEl.textContent = errorMsg || '';
  document.getElementById('auth-input').value = '';
  document.getElementById('auth-input').focus();
}

function hideAuthOverlay() {
  document.getElementById('auth-overlay').classList.add('hidden');
}

function signOut() {
  localStorage.clear();
  location.reload();
}

// Show/hide password toggle
document.getElementById('auth-eye').addEventListener('click', function() {
  const inp = document.getElementById('auth-input');
  if (inp.type === 'password') {
    inp.type = 'text';
    this.innerHTML = '&#128065;&#65039;';
  } else {
    inp.type = 'password';
    this.innerHTML = '&#128065;';
  }
});

// Submit overlay on button click or Enter key
document.getElementById('auth-submit').addEventListener('click', handleAuthSubmit);
document.getElementById('auth-input').addEventListener('keydown', function(e) {
  if (e.key === 'Enter') handleAuthSubmit();
});

function handleAuthSubmit() {
  const val = document.getElementById('auth-input').value.trim();
  if (!val) return;
  localStorage.setItem('api_key', val);
  hideAuthOverlay();
  // Re-trigger init calls now that we have a key
  loadDocs();
}
```
